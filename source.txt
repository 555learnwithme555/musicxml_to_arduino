/*
  Play melody
*/

#include "pitches.h"

!!MELODY!!

void setup() {
}

void loop() {
  for (int thisNote = 0; thisNote < melody_size + 1; thisNote++) {

    int noteDuration = durations[thisNote];
    tone(8, notes[thisNote],noteDuration);

    int pauseBetweenNotes = noteDuration * 1.30;
    delay(pauseBetweenNotes);
    noTone(8);
  }

  noTone(8);
  
  delay (100);
}
